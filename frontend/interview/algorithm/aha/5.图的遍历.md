图的邻接矩阵

1.深度优先搜索遍历图

二维数组e存储的是图的邻接矩阵，数组book记录哪些顶点已经访问过，变量sum用来记录已经访问过多少个顶点，变量n存储的是图的顶点的总个数。

```javascript
var book = [];
for(var i = 0; i < n; i++) {
    book[i] = 0;
}
var sum = 0, n = 5;
var e = [
    [0,1,1,0,1],
    [1,0,0,1,0],
    [1,0,0,0,1],
    [0,1,0,0,0],
    [1,0,1,0,0]
];
function dfs(cur) {  // cur是当前所在的顶点编号
    console.log(cur);
    sum++;
    if(sum == n) return;
    for (var i=0; i<n; i++) {
        if(e[cur][i] == 1 && book[i] == 0) {
            book[i] = 1; // 标记顶点已经访问过
            dfs(i);
        }
    }
    return;
}
dfs(0)
```
